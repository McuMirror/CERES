245 /*********************************************************************
246  * Function: BYTE getLCDAddressCounter(void)
247  *
248  * PreCondition: None
249  *
250  * Input: None
251  *
252  * Output: le compteur d'adresse.
253  *
254  * Side Effects: None
255  *
256  * Overview: None
257  *
258  * Note: Lit le compteur d'adresse de la ram
259  * Le bit de poid fort indique si l'afficheur est occcupé par
260  * une opération interne.
261  *
262  ********************************************************************/
263 BYTE getLCDAddressCounter()
264 {
265  BYTE byResult;
266 
267  PORTClearBits(LCD_RS); //RS à l'état bas.
268  PORTSetBits(LCD_RW); //RW en lecture.
269  PORTSetBits(LCD_EN); //Activé.
270  Delay10us(1);
271  PORTSetPinsDigitalIn(LCD_DATA);
272  byResult = PORTReadBits(LCD_DATA); //Lecture de la partie haute de l'adresse
273  PORTClearBits(LCD_EN); //Etat bas pour Enable
274  Delay10us(1);
275  PORTSetBits(LCD_EN); //Activé
276  byResult += PORTReadBits(LCD_DATA) >> 4; //Lecture de la partie basse de l'adresse.
277  Delay10us(1);
278  PORTClearBits(LCD_EN); //Etat bas pour Enable.
279  PORTClearBits(LCD_RW); //Sort du mode lecture.
280  LCD_RESET_PORT_DATA(); //Le port data à l'état bas
281  PORTSetPinsDigitalOut(LCD_DATA);
282  return byResult;
283 }
