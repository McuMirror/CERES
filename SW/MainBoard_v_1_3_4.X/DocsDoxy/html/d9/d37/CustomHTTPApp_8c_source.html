528 /*****************************************************************************
529 Function:
530 static HTTP_IO_RESULT HTTPPostLCD(void)
531 
532 Summary:
533 Processes the LCD form on forms.htm
534 
535 Description:
536 Locates the 'lcd' parameter and uses it to update the text displayed
537 on the isard's LCD display.
538 
539 This function has four states. The first reads a name from the data
540 string returned as part of the POST request. If a name cannot
541 be found, it returns, asking for more data. Otherwise, if the name
542 is expected, it reads the associated value and writes it to the LCD.
543 If the name is not expected, the value is discarded and the next name
544 parameter is read.
545 
546 In the case where the expected string is never found, this function
547 will eventually return HTTP_IO_NEED_DATA when no data is left. In that
548 case, the HTTP2 server will automatically trap the error and issue an
549 Internal Server Error to the browser.
550 
551 Precondition:
552 None
553 
554 Parameters:
555 None
556 
557 Return Values:
558 HTTP_IO_DONE - the parameter has been found and saved
559 HTTP_IO_WAITING - the function is pausing to continue later
560 HTTP_IO_NEED_DATA - data needed by this function has not yet arrived
561  ***************************************************************************/
562  #if defined(USE_LCD)
563 
564 static HTTP_IO_RESULT HTTPPostLCD(void)
565 {
566  BYTE* cDest;
567 
568  #define SM_POST_LCD_READ_NAME (0u)
569  #define SM_POST_LCD_READ_VALUE (1u)
570 
571  switch(curHTTP.smPost)
572  {
573  // Find the name
574  case SM_POST_LCD_READ_NAME:
575 
576  // Read a name
577  if(HTTPReadPostName(curHTTP.data, HTTP_MAX_DATA_LEN) == HTTP_READ_INCOMPLETE)
578  return HTTP_IO_NEED_DATA;
579 
580  curHTTP.smPost = SM_POST_LCD_READ_VALUE;
581  // No break...continue reading value
582 
583  // Found the value, so store the LCD and return
584  case SM_POST_LCD_READ_VALUE:
585 
586  // If value is expected, read it to data buffer,
587  // otherwise ignore it (by reading to NULL)
588  if(!strcmppgm2ram((char*)curHTTP.data, (ROM char*)"lcd"))
589  cDest = curHTTP.data;
