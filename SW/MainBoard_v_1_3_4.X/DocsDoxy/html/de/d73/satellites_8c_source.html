29 static BOOL isReceiveAnswer(BYTE byNumMachine, BYTE byNewNumMachine)
30 {
31  WORD wCRC;
32  SETREAD();
33  while(!UARTReceivedDataIsAvailable(UART485) && wTORequest);

36  return FALSE;

43  {
44  return FALSE;
45  }

57  return FALSE;

69 static BOOL isSendCommand(const BYTE byNumMachine, const BYTE byHeader,
70  const BYTE byLenData, const void *byData)
71 {
72  WORD wLCRC;
73  BYTE byRetry = 4;
74  BYTE byNewMachine = 0XFF;
75  do
76  {
77  DelayMs(20);
78  uartReg[UART485]->sta.reg = 0;
79  uartReg[UART485]->sta.set = _U1STA_UTXEN_MASK;
80  uartReg[UART485]->sta.set = _U1STA_URXEN_MASK;
81  memset(bySatBuffer, 0, sizeof (bySatBuffer));
82  bySatBuffer[0] = byNumMachine;
83  bySatBuffer[1] = byLenData;
84  bySatBuffer[2] = HOST;
85  bySatBuffer[3] = byHeader;
86  memmove(&bySatBuffer[4], byData, byLenData);
87  wLCRC = wCRC16(bySatBuffer, byLenData + 4);
88  memmove(&bySatBuffer[byLenData + 4], &wLCRC, sizeof (WORD));
89  vTaskSuspendAll();
90  RS485WriteBuffer(bySatBuffer, byLenData + 6);
91  xTaskResumeAll();
92  wTORequest = SAT_TO;
93  if(byHeader == MODIFY_MACHINE_NUMBER)
94  {
95  byNewMachine = bySatBuffer[4];
96  }
97  memset(bySatBuffer, 0, sizeof (bySatBuffer));
98  }while(!isReceiveAnswer(byNumMachine, byNewMachine) && --byRetry);
99  return (byRetry > 0);
100 }
